const cylinder = document.getElementById('cylinder'); const panels = Array.from(document.querySelectorAll('.panel')); const counter = document.getElementById('counter'); // 3D Math Settings const COLS = 7; const thetaY = 360 / COLS; const thetaX = 90; // 4 rows: Main, Proj1, Proj2, Proj3 let currentAngleY = 0; let currentAngleX = 0; let velocityY = 0; let velocityX = 0; let radius = 0; function initGeometry() { const w = cylinder.offsetWidth; const gap = window.innerWidth < 768 ? 60 : 150; radius = Math.round((w / 2) / Math.tan(Math.PI / COLS)) + gap; panels.forEach((panel) => { const col = parseInt(panel.getAttribute('data-col') || 0); const row = parseInt(panel.getAttribute('data-row') || 0); panel.style.transform = `rotateY(${col * thetaY}deg) rotateX(${-row * thetaX}deg) translateZ(${radius}px)`; }); } function loop() { // Friction velocityY *= 0.95; velocityX *= 0.95; // Snapping logic for Y if (Math.abs(velocityY) < 2) { const snapY = Math.round(currentAngleY / thetaY) * thetaY; const distY = currentAngleY - snapY; if (Math.abs(distY) < (thetaY * 0.25)) { velocityY -= distY * 0.15; velocityY *= 0.5; } } // Snapping logic for X if (Math.abs(velocityX) < 2) { const snapX = Math.round(currentAngleX / thetaX) * thetaX; const distX = currentAngleX - snapX; if (Math.abs(distX) < (thetaX * 0.25)) { velocityX -= distX * 0.15; velocityX *= 0.5; } } currentAngleY += velocityY; currentAngleX += velocityX; // Apply Torus Transform cylinder.style.transform = `translateZ(${-radius}px) rotateX(${currentAngleX}deg) rotateY(${currentAngleY}deg)`; // Update counter based on column const nearestCol = Math.round(-currentAngleY / thetaY); const realCol = ((nearestCol % COLS) + COLS) % COLS; counter.textContent = String(realCol + 1).padStart(2, '0') + ' / ' + String(COLS).padStart(2, '0'); requestAnimationFrame(loop); } function goToCol(col) { // Oblicz docelowy kąt tak by uwzględnić obecne obroty (najkrótsza droga) const snapY = Math.round(currentAngleY / thetaY) * thetaY; currentAngleY = -col * thetaY; velocityY = 0; // Zawsze wracamy do wiersza 0 przy zmianie kolumny currentAngleX = 0; velocityX = 0; } function goToRow(row) { currentAngleX = row * thetaX; velocityX = 0; } // ── Touch / Drag ── let touchStartX = 0; let touchStartY = 0; let isDragging = false; window.addEventListener('touchstart', (e) => { touchStartX = e.touches[0].clientX; touchStartY = e.touches[0].clientY; isDragging = true; }, {passive: true}); window.addEventListener('touchmove', (e) => { if (!isDragging) return; let deltaX = e.touches[0].clientX - touchStartX; let deltaY = e.touches[0].clientY - touchStartY; if (Math.abs(deltaX) > Math.abs(deltaY)) { // Horizontal pan currentAngleY += deltaX * 0.05; velocityY = deltaX * 0.05; } else { // Vertical pan // Allow vertical pan only if we are on Evidence (col=1) const nearestCol = Math.round(-currentAngleY / thetaY); const realCol = ((nearestCol % COLS) + COLS) % COLS; if (realCol === 1) { // To rotate up, we increase angleX when pulling down currentAngleX += deltaY * 0.05; velocityX = deltaY * 0.05; } } touchStartX = e.touches[0].clientX; touchStartY = e.touches[0].clientY; }, {passive: false}); window.addEventListener('touchend', (e) => { isDragging = false; }); // ── Resize ── let rt; window.addEventListener('resize', () => { clearTimeout(rt); rt = setTimeout(initGeometry, 100); }); // ── Init ── initGeometry(); loop();
